3   public class Ejercicio4 {
4   	private double[] notas;
5   	private String[] calificaciones;
6   
7   	public void calculaCalificaciones(double[] notas) {
8   		String calificacion = "";
9   		int i = 0;
10  		this.notas = notas;
11  		this.calificaciones = new String[notas.length];
12  		for (double nota : notas) {
13  			calificacion = obtenerCalificacion(nota);
14  			this.calificaciones[i] = calificacion;
15  			i++;
16  		}
17  	}
18  	
19  	public static String obtenerCalificacion(double nota) {
20  		if (nota >= 0.0 && nota < 5.0)
21  			return "Suspenso";
22  		else if (nota >= 5.0 && nota < 7.0)
23  			return "Aprobado";
24  		else if (nota >= 7.0 && nota < 9.0)
25  			return "Notable";
26  		else if (nota >= 9.0 && nota < 10.0)
27  			return "Sobresaliente";
28  		else if (nota == 10.0)
29  			return "Matricula";
30  		else
31  			return "Error en la nota";
32  	}
33  
34  	public String[] getCalificaciones() {
35  		return calificaciones;
36  	}
37  	
38  	public double[] getNotas() {
39  		return notas;
40  	}
